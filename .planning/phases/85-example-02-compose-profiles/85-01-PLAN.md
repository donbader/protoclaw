1. **Add `StringOrArray` type** in `types.rs` above `LocalWorkspaceConfig`:

```rust
/// A value that accepts either a single string or an array of strings in YAML.
/// Single strings are normalized to a one-element Vec for uniform handling.
/// Used for `binary` and `entrypoint` fields that now accept command + args.
#[derive(Debug, Clone, PartialEq)]
pub struct StringOrArray(pub Vec<String>);

impl StringOrArray {
    pub fn as_slice(&self) -> &[String] { &self.0 }
    pub fn first(&self) -> Option<&str> { self.0.first().map(|s| s.as_str()) }
    pub fn command_and_args(&self) -> (&str, &[String]) {
        let (cmd, args) = self.0.split_first().expect("StringOrArray must have at least one element");
        (cmd.as_str(), args)
    }
}
```

Implement custom `Deserialize` (try sequence first via `serde_json::Value`, fall back to string) and `Serialize` (single element → string, multiple → array). Use serde untagged-style manual impl:

```rust
impl<'de> Deserialize<'de> for StringOrArray {
    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
    where D: Deserializer<'de> {
        let value = serde_json::Value::deserialize(deserializer)?;
        match value {
            serde_json::Value::String(s) => Ok(StringOrArray(vec![s])),
            serde_json::Value::Array(arr) => {
                let strings: Result<Vec<String>, _> = arr.into_iter()
                    .map(|v| v.as_str().map(|s| s.to_string())
                        .ok_or_else(|| serde::de::Error::custom("array elements must be strings")))
                    .collect();
                let strings = strings?;
                if strings.is_empty() {
                    return Err(serde::de::Error::custom("array must not be empty"));
                }
                Ok(StringOrArray(strings))
            }
            _ => Err(serde::de::Error::custom("expected string or array of strings")),
        }
    }
}

impl Serialize for StringOrArray {
    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
    where S: serde::Serializer {
        if self.0.len() == 1 {
            self.0[0].serialize(serializer)
        } else {
            self.0.serialize(serializer)
        }
    }
}
```
