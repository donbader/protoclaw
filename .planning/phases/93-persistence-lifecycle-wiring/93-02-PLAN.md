```rust
async fn create_session(&mut self, agent_name: &str, session_key: SessionKey) -> Result<String, AgentsError> {
    let slot_idx = find_slot_by_name(&self.slots, agent_name)
        .ok_or_else(|| AgentsError::AgentNotFound(agent_name.to_string()))?;
    let slot = &self.slots[slot_idx];
    if let Some(acp_id) = slot.session_map.get(&session_key) {
        return Ok(acp_id.clone());
    }
    let acp_timeout = Self::acp_timeout_for(&self.slots[slot_idx].config, &self.manager_config);
    let acp_session_id = Self::start_session(&mut self.slots[slot_idx], &self.tools_handle, acp_timeout).await?;
    let slot = &mut self.slots[slot_idx];
    slot.session_map.insert(session_key.clone(), acp_session_id.clone());
    slot.reverse_map.insert(acp_session_id.clone(), session_key);
    tracing::info!(agent = %agent_name, session_key = %acp_session_id, "multi-session created");
    Ok(acp_session_id)
}
```

**In create_session() (line ~447), after the session_map.insert + reverse_map.insert block and before the tracing::info, add:**

```rust
// Persist to store (fire-and-forget on failure — session works in-memory regardless)
let now = std::time::SystemTime::now()
    .duration_since(std::time::UNIX_EPOCH)
    .map(|d| d.as_secs() as i64)
    .unwrap_or(0);
let persisted = protoclaw_core::PersistedSession {
    session_key: session_key.to_string(),
    agent_name: agent_name.to_string(),
    acp_session_id: acp_session_id.clone(),
    created_at: now,
    last_active_at: now,
    closed: false,
};
if let Err(e) = self.session_store.upsert_session(&persisted).await {
    tracing::warn!(
        agent = %agent_name,
        session_key = %session_key,
        error = %e,
        "failed to persist new session to store"
    );
}
```

**In prompt_session() (line ~473), after the `send_request("session/prompt", params)` succeeds (line ~502) but before spawning the completion task, add:**

```rust
// Update last-active timestamp in store
let now = std::time::SystemTime::now()
    .duration_since(std::time::UNIX_EPOCH)
    .map(|d| d.as_secs() as i64)
    .unwrap_or(0);
let sk_string = session_key.to_string();
let store = self.session_store.clone();
// Fire-and-forget — don't block the prompt on store I/O
tokio::spawn(async move {
    if let Err(e) = store.update_last_active(&sk_string, now).await {
        tracing::warn!(
            session_key = %sk_string,
            error = %e,
            "failed to update last_active in store"
        );
    }
});
```

Note: The `update_last_active` call is spawned as a background task so it doesn't add latency to the prompt path. The `session_store` is `Arc<dyn DynSessionStore>` so cloning is cheap.
