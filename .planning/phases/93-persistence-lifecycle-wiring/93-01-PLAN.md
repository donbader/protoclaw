   ```rust
   fn build_session_store(config: &SessionStoreConfig) -> std::sync::Arc<dyn protoclaw_core::DynSessionStore> {
       match config {
           SessionStoreConfig::None => {
               std::sync::Arc::new(protoclaw_core::NoopSessionStore)
           }
           SessionStoreConfig::Sqlite(sqlite_cfg) => {
               let store = match &sqlite_cfg.path {
                   Some(path) => protoclaw_core::SqliteSessionStore::open(path),
                   None => protoclaw_core::SqliteSessionStore::open_in_memory(),
               };
               match store {
                   Ok(s) => std::sync::Arc::new(s),
                   Err(e) => {
                       tracing::error!(error = %e, "failed to open session store, falling back to noop");
                       std::sync::Arc::new(protoclaw_core::NoopSessionStore)
                   }
               }
           }
       }
   }
   ```

6. In `shutdown_all()` (line ~1081), before sending `session/close` notifications, call `mark_closed` for each session:
   ```rust
   async fn shutdown_all(&mut self) {
       for slot in &mut self.slots {
           // Mark all sessions closed in the persistent store BEFORE sending close to agent
           for session_key in slot.session_map.keys() {
               if let Err(e) = self.session_store.mark_closed(&session_key.to_string()).await {
                   tracing::warn!(session_key = %session_key, error = %e, "failed to mark session closed in store");
               }
           }
           if let Some(conn) = &slot.connection {
               for acp_id in slot.session_map.values() {
                   let params = serde_json::json!({ "sessionId": acp_id });
                   let _ = conn.send_notification("session/close", params).await;
               }
               tokio::time::sleep(Duration::from_millis(self.manager_config.shutdown_grace_ms))
                   .await;
           }
           if let Some(mut conn) = slot.connection.take() {
               let _ = conn.kill().await;
           }
       }
   }
   ```

   Note: `self.session_store` is `Arc<dyn DynSessionStore>` which is not `&mut self` — it's cloneable and callable from `&self`. But `shutdown_all` takes `&mut self`. The `session_store` field is `Arc` so calling `.mark_closed()` only needs `&self` on the Arc. This works because `shutdown_all(&mut self)` has exclusive access, and `Arc::mark_closed` takes `&self`.

   }
